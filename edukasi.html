sephon32.ino 
#include <WiFi.h>
#include <Sephon32.h>

void setup() {
  Sephon.begin();
  Sephon.connectWifi(ssid, pass);
}

void loop() {
  float pH  = Sephon.readPH();
  float ntu = Sephon.readTurb();
  float tmp = Sephon.readTemp();
  Sephon.sendData(pH, ntu, tmp);
}
